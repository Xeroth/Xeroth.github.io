实例:

1.去重复行, 原理就是利用初始数组变量值为零去重原理分析

awk '!a[$0]++' <filename>

2.显示所有只有四个字符的名

awk  '{if(length($1)==4)  print $1}' datafile

3.文件合并

我有2个文件 tmp1.txt 和 tmp2.txt

tmp1.txt
20
18
19

tmp2.txt
15
14
13

我想做个文件tmp3.txt做成

20 15
18 14
19 13

方法一, 用 paste 命令

paste tmp1.txt tmp2.txt >tmp3.txt

方法二, 用 FNR 和 NR , 输出$0 (合并2个文件有用)

awk 'NR==FNR{a[NR]=$0} NR>FNR{print a[FNR],$0}' tmp1.txt tmp2.txt | tee tmp3.txt

方法三, 使用数组存储$1

awk 'NR==FNR{a[NR]=$1;count=NR} NR>FNR{b[NR-count]=$1} END{for(i=1;i<=count;i++)print a[i],b[i]}' tmp1.txt tmp2.txt | tee tmp3.txt

4.把以\结束的行同下一行相接(sub, getline, next用法)

awk '/\\$/ { sub(/\\$/,""); getline t; print $0 t; next }; 1′

首先查找以\结束的行，并删除\。然后getline函数获取下一行的内容，输出当前行（去掉了\）和下一行的相接后的内容。并用next跳过后面的1避免重复输出。如果当前行行末没有\，会直接执行1输出。

5.合并行输出(next在多行合并，以及选择性输出方面，非常方便)

6.把每5行用逗号相连

pattern ? pattern : pattern (三目运算符) 满足表达式条件, 就输出?后面的, 不满足输出:后面的

awk 'ORS=NR%5?",":"\n"'

这里在每行输出前，对ORS重新进行定义，如果行号能被5整除则为\n，否则为逗号。（也很天书的一句）

7.getline 的一些用法

#输出偶数行, 不跟变量赋值给$0
seq 10 |awk '{getline;print}'
#输出奇数行
seq 10 |awk '{getline t;print}'
#配合 system , 因为 system(cmd)返回的是 cmd 的返回值
awk 'BEGIN{ while( system("ls -l") | getline line ){ print line } }'

8.split

The awk function split(s,a,sep) splits a string s into an awk array a using the delimiter sep.

set time = 12:34:56
set hr = `echo $time | awk '{split($0,a,":" ); print a[1]}'` # = 12
set sec = `echo $time | awk '{split($0,a,":" ); print a[3]}'` # = 56
# = 12 34 56
set hms = `echo $time | awk '{split($0,a,":" ); print a[1], a[2], a[3]}'`

Q:

name="76868&5676&435&43526&334&12312312&12321"
awk 'BEGIN {print split("$name", filearray, "&")}'

为什么是1

而

awk 'BEGIN {print split("76868&5676&435&43526&334&12312312&12321", filearray, "&")}'

则返回正确的结果，应该是7，有没有人解答一下？

A:

变量引用错误，这样做试试

awk 'BEGIN {print split('"\"$name\""', filearray, "&")}'

awk规定引用系统变量必须使用单引号加双引号，即’”$sysvar”‘这样的格式，但是split函数也需要双引号来定界，但这个双引号又不能让sh解释，而应留给awk来解释，所以使用了\”和\”组成的双引号
